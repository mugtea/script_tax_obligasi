CREATE or alter view [dbo].vw_TaxCustomerDeductedAdvice_OBS_ObligasiSummary_Header
as
	SELECT distinct
  trl.period_start  TanggalAwalPelaporan,
  trl.period_end  TanggalAkhirPelaporan,
  osh.npwp npwp,
  osh.CIF CIF,
  osh.NamaNasabah NamaNasabah,
  trl.id
FROM (select * from (select ROW_NUMBER() over (partition by cif order by dataid desc) RowID, npwp, CIF,NamaNasabah from TaxCustomerDeductedAdvice_ObligasiSummary_History) x where x.RowID = 1) osh
join TaxCustomerDeductedAdvice_ObligasiSummary_CustomerInformation osci on osh.CIF = osci.CIF
join TaxRequestList trl on trl.id = osci.Fk_TaxRequestList
GO
