CREATE OR ALTER PROCEDURE [dbo].[TaxCustomerDeductedAdvice_OBS_SetGeneratedFile]
as
begin
update trl set is_generated = 1 From FileData fd
left join MappingStatementFileData msf on msf.Fk_FileData_Id = fd.Pk_FileData_Id
left join Statement s on msf.Fk_Statement_Id = s.Pk_Statement_Id
left join TaxRequestList trl on trl.id = s.FK_TaxRequestList
where 
--trl.period_type = 'Y' and
 trl.is_generated is null
end
